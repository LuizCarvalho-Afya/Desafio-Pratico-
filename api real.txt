import requests
import heapq

def get_pokemon_data(name):
    url_species = f"https://pokeapi.co/api/v2/pokemon-species/{name.lower()}"
    res = requests.get(url_species).json()
    evo_chain_url = res['evolution_chain']['url']
    evo_data = requests.get(evo_chain_url).json()['chain']
    
    edges = []
    def build_graph(node):
        for evo in node['evolves_to']:
            u = node['species']['name']
            v = evo['species']['name']
            # Custo: nível mínimo ou 1 para métodos especiais
            w = evo['evolution_details'][0].get('min_level') or 1
            edges.append((u, v, w))
            build_graph(evo)
            
    build_graph(evo_data)
    return edges, evo_data['species']['name']

    for u, v, w in edges:
        adj[u].append((v, w))
        nodes.add(u); nodes.add(v)
    
    pq = [(0, start_node)]
    dist = {n: float('inf') for n in nodes}
    dist[start_node] = 0
    
    while pq:
        d, u = heapq.heappop(pq)
        if d > dist[u]: continue
        for v, w in adj[u]:
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                heapq.heappush(pq, (dist[v], v))
    return dist
